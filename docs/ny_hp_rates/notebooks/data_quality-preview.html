This notebook checks the quality of gas bill data in the NY HP rates master tables. It compares gas bills before (upgrade 0) and after (upgrade 2) heat pump installation, broken down by gas tariff transition, to verify that the gas calc fix (ny_20260307_r1-8_gascalcfix batch) is producing sensible results.

Median bill, cost-of-service, and bill alignment by customer group

A high-level sanity check: for each customer group in the baseline scenario (run 1+2), what is the weighted median annual bill, cost-of-service (marginal cost + residual), and bill alignment (BAT)?

Cost-of-service = economic_burden + residual_share from the BAT table. BAT = bill − cost-of-service. A positive BAT means the customer overpays relative to their cost of service; a negative BAT means they underpay.

def _weighted_median(df: pl.DataFrame, col: str, w_col: str = "weight") -> float:
    s = df.sort(col)
    cum = s[w_col].cum_sum() / s[w_col].sum()
    return float(s.filter(cum >= 0.5)[col][0])


GROUPS = {
    "HP customers": bat.filter(pl.col("postprocess_group.has_hp")),
    "Non-HP customers": bat.filter(~pl.col("postprocess_group.has_hp")),
    "Fossil fuel customers": bat.filter(
        pl.col("heats_with_natgas") | pl.col("heats_with_oil") | pl.col("heats_with_propane")
    ),
    "Natural gas customers": bat.filter(pl.col("heats_with_natgas")),
    "Oil/propane customers": bat.filter(pl.col("heats_with_oil") | pl.col("heats_with_propane")),
    "Electric resistance customers": bat.filter(pl.col("postprocess_group.heating_type") == "electrical_resistance"),
}

summary_rows = []
for name, g in GROUPS.items():
    def _wmean(col: str) -> float:
        return (g[col] * g["weight"]).sum() / g["weight"].sum()

    summary_rows.append({
        "Group": name,
        "N (weighted)": round(g["weight"].sum()),
        "Med. Bill (delivery)": round(_weighted_median(g, "annual_bill_delivery")),
        "Med. CoS (delivery)": round(_weighted_median(g, "cos_delivery")),
        "Med. BAT (delivery)": round(_weighted_median(g, "BAT_percustomer_delivery")),
        "Mean BAT (delivery)": round(_wmean("BAT_percustomer_delivery")),
        "Med. Bill (total)": round(_weighted_median(g, "annual_bill_total")),
        "Med. CoS (total)": round(_weighted_median(g, "cos_total")),
        "Med. BAT (total)": round(_weighted_median(g, "BAT_percustomer_total")),

Build comparison table

We join the two upgrade scenarios on bldg_id, restrict to natgas-heated buildings with nonzero gas bills in u0, and attach the gas tariff keys from both upgrade maps.

In [10]:
merged = (
    bills_u0.filter(pl.col("heats_with_natgas"))
    .select("bldg_id", "sb.electric_utility", pl.col("gas_total_bill").alias("gas_u0"))
    .join(
        bills_u2.select("bldg_id", pl.col("gas_total_bill").alias("gas_u2")),
        on="bldg_id",
    )
    .filter(pl.col("gas_u0") > 0)
    .with_columns(
        (pl.col("gas_u2") - pl.col("gas_u0")).alias("gas_diff"),
        ((pl.col("gas_u2") - pl.col("gas_u0")) / pl.col("gas_u0") * 100).alias("gas_pct"),
    )
    .join(tariff_u0, on="bldg_id", how="left")
    .join(tariff_u2, on="bldg_id", how="left")
    .join(gas_u0, on="bldg_id", how="left")
    .join(gas_u2, on="bldg_id", how="left")
    .with_columns(
        ((pl.col("gas_kwh_u2") - pl.col("gas_kwh_u0")) / pl.col("gas_kwh_u0") * 100)
        .alias("cons_pct"),
    )
)
print(f"{merged.height} natgas-heated buildings with nonzero gas in u0")

Gas bill change by tariff transition

Each row shows how a group of buildings — defined by their gas tariff in u0 and u2 — fares after the heat pump upgrade. We expect gas bills to decrease for most transitions since gas consumption drops ~89% after switching to a heat pump.

#| label: histogram-helpers
BIN_WIDTH = 100
QUADRANT_COLORS = {
    "savings > $1k": "#1b5e20",
    "savings $0-1k": "#81c784",
    "losses $0-1k": "#ef9a9a",
    "losses > $1k": "#b71c1c",
}
QUADRANT_ORDER = list(QUADRANT_COLORS.keys())


def _weighted_percentile(df: pl.DataFrame, col: str, w_col: str, q: float) -> float:
    """Weighted percentile via cumulative weight."""
    s = df.sort(col)
    cum = s[w_col].cum_sum() / s[w_col].sum()
    return float(s.filter(cum >= q)[col][0])


def quadrant_pcts(df: pl.DataFrame) -> dict[str, float]:
    """Weighted percentage of homes in each bill-change quadrant."""
    total = df["weight"].sum()
    return {
        "savings > $1k": df.filter(pl.col("delta") < -1000)["weight"].sum() / total * 100,
        "savings $0-1k": df.filter((pl.col("delta") >= -1000) & (pl.col("delta") < 0))["weight"].sum() / total * 100,
        "losses $0-1k": df.filter((pl.col("delta") >= 0) & (pl.col("delta") < 1000))["weight"].sum() / total * 100,
        "losses > $1k": df.filter(pl.col("delta") >= 1000)["weight"].sum() / total * 100,
    }


def plot_bill_change_histogram_faceted(
    df: pl.DataFrame,
    facet_col: str,
    facet_order: list[str],
    bin_width: int = BIN_WIDTH,
) -> ggplot:
    """Weighted histogram of annual bill change, faceted by a categorical column."""
    snap = max(bin_width, 50)
    x_lo = math.floor(_weighted_percentile(df, "delta", "weight", 0.01) / snap) * snap
    x_hi = math.ceil(_weighted_percentile(df, "delta", "weight", 0.99) / snap) * snap

    facet_dtype = pl.Enum(facet_order)
    quadrant_dtype = pl.Enum(QUADRANT_ORDER)

    binned = (
        df.with_columns(
            ((pl.col("delta") / bin_width).floor() * bin_width + bin_width / 2).alias("bin_center"),
        )
        .with_columns(
            pl.when(pl.col("bin_center") <= -1000)
            .then(pl.lit("savings > $1k"))
            .when((pl.col("bin_center") > -1000) & (pl.col("bin_center") < 0))
            .then(pl.lit("savings $0-1k"))
            .when((pl.col("bin_center") >= 0) & (pl.col("bin_center") < 1000))
            .then(pl.lit("losses $0-1k"))
            .otherwise(pl.lit("losses > $1k"))
            .alias("quadrant"),
        )
        .group_by("bin_center", "quadrant", facet_col)
        .agg(pl.col("weight").sum().alias("weight_sum"))
        .with_columns(
            pl.col("quadrant").cast(quadrant_dtype),
            pl.col(facet_col).cast(facet_dtype),
        )
    )

    annot_rows: list[dict] = []
    quadrant_bounds = [(x_lo, -1000), (-1000, 0), (0, 1000), (1000, x_hi)]
    label_xs = [(max(lo, x_lo) + min(hi, x_hi)) / 2 for lo, hi in quadrant_bounds]

    for fv in facet_order:
        facet_df = df.filter(pl.col(facet_col) == fv)
        if facet_df.is_empty():
            continue
        pct = quadrant_pcts(facet_df)
        facet_binned = binned.filter(pl.col(facet_col) == fv)
        max_w = float(facet_binned["weight_sum"].max()) if not facet_binned.is_empty() else 0.0
        y_val = max_w * 1.08
        for label, x_pos in zip(QUADRANT_ORDER, label_xs):
            if pct[label] < 0.5:
                continue
            annot_rows.append({
                facet_col: fv,
                "x": x_pos,
                "y": y_val,
                "label_text": f"{label}\n{pct[label]:.0f}%",
                "color_hex": QUADRANT_COLORS[label],
            })

    annot_df = pl.DataFrame(annot_rows).with_columns(
        pl.col(facet_col).cast(facet_dtype),
    )

    p = (
        ggplot(binned, aes(x="bin_center", y="weight_sum", fill="quadrant"))
        + geom_col(width=bin_width * 0.9)
        + geom_vline(xintercept=[-1000, 0, 1000], linetype="dotted", color="gray")
        + scale_fill_manual(values=QUADRANT_COLORS, breaks=QUADRANT_ORDER)
        + coord_cartesian(xlim=(x_lo, x_hi))
        + facet_wrap(facet_col, ncol=1, scales="free_y")
        + labs(x="Annual Bill Change ($)", y="# of Homes")
        + guides(fill=False)
        + theme_switchbox()
        + theme(
            figure_size=(10, 8),
            strip_text=element_text(size=11, weight="bold"),
        )
    )

    if not annot_df.is_empty():
        p = (
            p
            + geom_text(
                annot_df,
                aes(x="x", y="y", label="label_text", color="color_hex"),
                inherit_aes=False,
                ha="center", size=9,
            )
            + scale_color_identity()
        )

    return p

Delivery bill percent change by cooling type and utility

The delivery bill (volumetric delivery + fixed charge) is the component most directly affected by the cross-subsidy problem. How much does it increase after switching to a heat pump? Homes with no baseline cooling see the largest delivery bill increases because they gain air conditioning on top of heating electrification.

#| label: compute-delivery-pct-change
delivery_pct_change = (
    bills_u0
    .select(
        BLDG_ID,
        (pl.col("elec_delivery_bill") + pl.col("elec_fixed_charge")).alias("delivery_before"),
        "heats_with_natgas", "sb.electric_utility", "weight",
    )
    .join(
        bills_u2.select(
            BLDG_ID,
            (pl.col("elec_delivery_bill") + pl.col("elec_fixed_charge")).alias("delivery_after"),
        ),
        on=BLDG_ID,
    )
    .filter(pl.col("heats_with_natgas"))
    .filter(pl.col("delivery_before").abs() > 1)
    .join(metadata_sb.select(BLDG_ID, COOLING_COL), on=BLDG_ID, how="left")
    .with_columns(
        ((pl.col("delivery_after") - pl.col("delivery_before")) / pl.col("delivery_before") * 100)
        .alias("pct_change"),
        pl.when(pl.col(COOLING_COL) == "100% Conditioned").then(pl.lit("Full cooling"))
        .when(pl.col(COOLING_COL) == "None").then(pl.lit("No cooling"))
        .otherwise(pl.lit("Partial cooling"))
        .alias("cooling_group"),
        pl.col("sb.electric_utility").replace_strict(UTIL_NAMES).alias("utility_label"),
    )
)

for cg in COOLING_GROUPS:
    _sub = delivery_pct_change.filter(pl.col("cooling_group") == cg)
    _med = _weighted_percentile(_sub, "pct_change", "weight", 0.50)
    print(f"{cg}: n={len(_sub)}, weighted median delivery bill pct change = {_med:.0f}%")
