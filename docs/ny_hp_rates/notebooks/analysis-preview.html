This notebook analyzes CAIRO simulation outputs for New York heat pump rate design. It produces two families of visualization:

  1. Bill change histograms — weighted distributions of annual total energy bill changes after switching to heat pumps, by population subgroup.
  2. Median electric bill component breakdown — stacked bar chart decomposing the weighted-median electric bill into supply, delivery volumetric, and delivery fixed charges for current HVAC vs heat pump scenarios.

#| label: bill-component-helpers
def bill_components_for_scenario(
    bills_lf: pl.LazyFrame,
    utility: str | None = None,
) -> pl.DataFrame:
    """Extract annual bill components (electric + gas + delivered fuels) from master bills."""
    filtered = bills_lf.filter(pl.col("month") == ANNUAL_MONTH)
    if utility is not None:
        filtered = filtered.filter(pl.col("sb.electric_utility") == utility)
    return (
        filtered
        .select(
            BLDG_ID, "weight",
            pl.col("elec_supply_bill").alias("supply"),
            pl.col("elec_delivery_bill").alias("delivery_volumetric"),
            pl.col("elec_fixed_charge").alias("delivery_fixed"),
            pl.col("gas_total_bill").alias("gas"),
            pl.col("propane_total_bill").alias("propane"),
            pl.col("oil_total_bill").alias("oil"),
            pl.col("energy_total_bill").alias("energy_total"),
            "heats_with_natgas",
        )
        .collect()
    )


def weighted_median_row(
    df: pl.DataFrame,
    sort_col: str = "energy_total",
    weight_col: str = "weight",
) -> pl.DataFrame:
    """Return the single row at the weighted median by sort_col."""
    total_weight = df[weight_col].sum()
    sorted_df = df.sort(sort_col)
    cum = sorted_df[weight_col].cum_sum()
    idx = (cum >= 0.5 * total_weight).arg_max()  # first True in a bool Series = first row past 50%
    assert idx is not None, "Empty data or non-positive weights"
    return sorted_df.slice(idx, 1)

def _format_dollar(v: float) -> str:
    return f"${v:,.0f}"


def _dollar_labels(df: pl.DataFrame) -> pl.DataFrame:
    return df.with_columns(
        pl.col("value")
        .map_elements(_format_dollar, return_dtype=pl.String)
        .alias("label"),
    )


def _totals_df(plot_df: pl.DataFrame) -> pl.DataFrame:
    return _dollar_labels(
        plot_df.group_by("scenario").agg(pl.col("value").sum())
    )


def _side_label_annotations(
    segments: list[tuple[str, float, float]],
    label_colors: dict[str, str],
) -> list:
    """Build annotate() layers for side-labels pointing at bar 2 segments.

    *segments* is a list of ``(label_text, bottom_y, height)`` tuples.
    """
    annotations: list = []
    for label_text, bottom, height in segments:
        y_mid = bottom + height / 2
        color_key = label_text.replace("\n", " ")
        color = label_colors[color_key]
        annotations.append(annotate(
            "segment",
            x=LINE_X, xend=LABEL_X - 0.03,
            y=y_mid, yend=y_mid,
            color=color, size=0.5,
        ))
        annotations.append(annotate(
            "text",
            x=LABEL_X, y=y_mid,
            label=label_text,
            color=color, ha="left", va="center",
            size=8, fontweight="bold",
        ))
    return annotations


# ── Simple chart (gas + electric, no decomposition) ──────────────────────────

def unpivot_median_simple(median_df: pl.DataFrame, scenario: str) -> pl.DataFrame:
    """Aggregate electric sub-components into one 'Electric Bill' row."""
    electric = float(
        median_df["supply"][0]
        + median_df["delivery_volumetric"][0]
        + median_df["delivery_fixed"][0]
    )
    gas = float(median_df["gas"][0])
    return pl.DataFrame({
        "component": ["Electric Bill", "Gas Bill"],
        "value": [electric, gas],
        "scenario": [scenario, scenario],
    })


def plot_bill_components_simple(
    med_current: pl.DataFrame,
    med_hp: pl.DataFrame,
    title: str = "Median energy bill components",
) -> ggplot:
    """Stacked bar: total electric vs gas for current HVAC and heat pump."""
    long = pl.concat([
        unpivot_median_simple(med_current, SCENARIO_ORDER[0]),
        unpivot_median_simple(med_hp, SCENARIO_ORDER[1]),
    ])

    scenario_dtype = pl.Enum(SCENARIO_ORDER)
    component_dtype = pl.Enum(SIMPLE_COMPONENT_ORDER)

    plot_df = _dollar_labels(long.with_columns(
        pl.col("scenario").cast(scenario_dtype),
        pl.col("component").cast(component_dtype),
    ))
    totals = _totals_df(plot_df)

    electric_hp = float(
        med_hp["supply"][0] + med_hp["delivery_volumetric"][0] + med_hp["delivery_fixed"][0]
    )
    gas_hp = float(med_hp["gas"][0])
    side_labels = _side_label_annotations(
        [("Electric Bill", 0, electric_hp), ("Gas Bill", electric_hp, gas_hp)],
        SIMPLE_LABEL_COLORS,
    )

    p = (
        ggplot(plot_df, aes(x="scenario", y="value", fill="component"))
        + geom_col(position=position_stack(reverse=True), width=0.6)
        + geom_text(
            aes(label="label"),
            position=position_stack(vjust=0.5, reverse=True),
            color="white", fontweight="bold", size=11,
        )
        + geom_text(
            aes(x="scenario", y="value", label="label"),
            data=totals,
            va="bottom", color="#333333", fontweight="bold",
            size=11, nudge_y=30, inherit_aes=False,
        )
        + scale_fill_manual(values=SIMPLE_COLORS)
        + scale_x_discrete(limits=SCENARIO_ORDER, expand=(0, 0.6, 0, 1.8))
        + scale_y_continuous(expand=(0, 0, 0.08, 0))
        + guides(fill=False)
        + labs(x="", y="Annual energy bill ($)", title=title)
        + theme_switchbox()
    )
    for sl in side_labels:
        p = p + sl
    return p


# ── Decomposed chart (electric sub-components + muted gas + delivery line) ───

def unpivot_median(median_df: pl.DataFrame, scenario: str) -> pl.DataFrame:
    """Unpivot a single-row median into long-form (scenario, component, value)."""
    cols = list(COMPONENT_NAMES.keys())
    return (
        median_df.select(cols)
        .unpivot(on=cols, variable_name="col", value_name="value")
        .with_columns(
            pl.col("col").replace_strict(COMPONENT_NAMES).alias("component"),
            pl.lit(scenario).alias("scenario"),
        )
        .drop("col")
    )


def plot_bill_components_decomposed(

BIN_WIDTH = 100
QUADRANT_COLORS = {
    "savings > $1k": "#1b5e20",
    "savings $0-1k": "#81c784",
    "losses $0-1k": "#ef9a9a",
    "losses > $1k": "#b71c1c",
}
QUADRANT_ORDER = list(QUADRANT_COLORS.keys())
QUADRANT_LABELS = {
    "losses > $1k": "LOSE > $1K",
    "losses $0-1k": "LOSE $0-1K",
    "savings $0-1k": "SAVE $0-1K",
    "savings > $1k": "SAVE > $1K",
}


def _weighted_percentile(df: pl.DataFrame, col: str, w_col: str, q: float) -> float:
    """Weighted percentile via cumulative weight."""
    s = df.sort(col)
    cum = s[w_col].cum_sum() / s[w_col].sum()
    return float(s.filter(cum >= q)[col][0])


def _weighted_mean(df: pl.DataFrame, col: str, w_col: str) -> float:
    """Weighted arithmetic mean."""
    return float((df[col] * df[w_col]).sum() / df[w_col].sum())


def quadrant_pcts(df: pl.DataFrame) -> dict[str, float]:
    """Weighted percentage of homes in each bill-change quadrant."""
    total = df["weight"].sum()
    return {
        "savings > $1k": df.filter(pl.col("delta") < -1000)["weight"].sum() / total * 100,
        "savings $0-1k": df.filter((pl.col("delta") >= -1000) & (pl.col("delta") < 0))["weight"].sum() / total * 100,
        "losses $0-1k": df.filter((pl.col("delta") >= 0) & (pl.col("delta") < 1000))["weight"].sum() / total * 100,
        "losses > $1k": df.filter(pl.col("delta") >= 1000)["weight"].sum() / total * 100,
    }


def plot_bill_change_histogram(
    df: pl.DataFrame,
    bin_width: int = BIN_WIDTH,
    title: str = "Change in Total Energy Bills after Switching to Heat Pumps",
) -> ggplot:
    """Weighted histogram of annual bill change."""
    snap = max(bin_width, 50)
    x_lo = math.floor(_weighted_percentile(df, "delta", "weight", 0.01) / snap) * snap
    x_hi = math.ceil(_weighted_percentile(df, "delta", "weight", 0.99) / snap) * snap

    binned = (
        df.with_columns(
            (
                (pl.col("delta") / bin_width).floor() * bin_width + bin_width / 2
            ).alias("bin_center"),
        )
        .with_columns(
            pl.when(pl.col("bin_center") <= -1000)
            .then(pl.lit("savings > $1k"))
            .when((pl.col("bin_center") > -1000) & (pl.col("bin_center") < 0))
            .then(pl.lit("savings $0-1k"))
            .when((pl.col("bin_center") >= 0) & (pl.col("bin_center") < 1000))
            .then(pl.lit("losses $0-1k"))
            .otherwise(pl.lit("losses > $1k"))
            .alias("quadrant"),
        )
        .group_by("bin_center", "quadrant")
        .agg(pl.col("weight").sum().alias("weight_sum"))
    )

    pct = quadrant_pcts(df)

    quadrant_dtype = pl.Enum(QUADRANT_ORDER)
    plot_df = binned.with_columns(pl.col("quadrant").cast(quadrant_dtype))
    max_weight = (
        cast(float, plot_df["weight_sum"].max()) if not plot_df.is_empty() else 0.0
    )
    y_annot = max_weight * 1.05

    quadrant_bounds = [
        (x_lo, -1000),
        (-1000, 0),
        (0, 1000),
        (1000, x_hi),
    ]
    label_xs = [
        (max(lo, x_lo) + min(hi, x_hi)) / 2 for lo, hi in quadrant_bounds
    ]

    annotations = []
    for label, x_pos in zip(QUADRANT_ORDER, label_xs):
        if pct[label] < 0.05:
            continue
        annotations.append(
            annotate(
                "text", x=x_pos, y=y_annot,
                label=f"{label}\n{pct[label]:.1f}%",
                ha="center", size=11, color=QUADRANT_COLORS[label],
            )
        )

    p = (
        ggplot(plot_df, aes(x="bin_center", y="weight_sum", fill="quadrant"))
        + geom_col(width=bin_width * 0.9)
        + geom_vline(xintercept=[-1000, 0, 1000], linetype="dotted", color="gray")
        + scale_fill_manual(values=QUADRANT_COLORS, breaks=QUADRANT_ORDER)
        + coord_cartesian(xlim=(x_lo, x_hi))
        + labs(
            x="Annual Bill Change ($)",
            y="# of Homes",
            title=title,
        )
        + guides(fill=False)
        + theme_switchbox()
        + theme(figure_size=(10.5, 4.5))
    )
    for a in annotations:
        p = p + a
    return p


def plot_quadrant_bar(
    rows: list[tuple[str, pl.DataFrame]],
    title: str = "How bills would change after switching to heat pumps:",
    title_parts: list[tuple[str, str]] | None = None,
) -> ggplot | Figure:
    """Horizontal stacked bar(s) showing quadrant proportions.

    *rows* is a list of ``(label, df)`` pairs where each *df* has ``delta``
    and ``weight`` columns. Each pair becomes one bar. Pass a single pair for
    one bar; pass multiple pairs for side-by-side comparison (e.g. different
    rate scenarios).

    When *title_parts* is given (a list of ``(text, color)`` tuples), the
    plotnine title is replaced by a matplotlib multi-color title and the
    function returns a ``Figure`` instead of a ``ggplot``.
    """
    records: list[dict] = []
    all_pcts: list[dict[str, float]] = []
    for row_label, df in rows:
        pct = quadrant_pcts(df)
        all_pcts.append(pct)
        for q in QUADRANT_ORDER:
            records.append({"scenario": row_label, "quadrant": q, "pct": pct[q]})

    scenario_order = [r[0] for r in reversed(rows)]
    plot_df = pl.DataFrame(records).with_columns(
        pl.col("quadrant").cast(pl.Enum(QUADRANT_ORDER)),
        pl.col("scenario").cast(pl.Enum(scenario_order)),
    )

    n_scenarios = len(rows)
    first_pct = all_pcts[0]
    stacking_order = list(reversed(QUADRANT_ORDER))
    extra_annotations: list = []

    cum = 0.0
    for q in stacking_order:
        mid = cum + first_pct[q] / 2
        cum += first_pct[q]
        if first_pct[q] < 2:
            continue
        extra_annotations.append(
            annotate(
                "text",
                x=n_scenarios + 0.45,
                y=mid,
                label=QUADRANT_LABELS[q],
                ha="center", va="bottom",
                size=8, color=QUADRANT_COLORS[q], fontweight="bold",
            )
        )

    if n_scenarios > 1:
        for i, (row_label, _) in enumerate(rows):
            bar_pos = n_scenarios - i
            extra_annotations.append(
                annotate(
                    "text",
                    x=bar_pos,
                    y=101,
                    label=row_label,
                    ha="left", va="center",
                    size=9, fontweight="bold",
                )
            )

    top_expand = 0.55 if n_scenarios > 1 else 1.2
    plotnine_title = "" if title_parts else title

    p = (
        ggplot(plot_df, aes(x="scenario", y="pct", fill="quadrant"))
        + geom_col(position="stack", width=0.55)
        + geom_text(
            plot_df.filter(pl.col("pct") >= 3),
            aes(label="pct"),
            position=position_stack(vjust=0.5),
            format_string="{:.0f}%",
            color="white",
            size=10,
            fontweight="bold",
        )
        + scale_fill_manual(values=QUADRANT_COLORS, breaks=QUADRANT_ORDER)
        + scale_y_continuous(expand=(0, 0, 0.15, 0))
        + scale_x_discrete(expand=(0, 0.35, 0, top_expand))
        + coord_flip()
        + guides(fill=False)
        + labs(x="", y="", title=plotnine_title)
        + theme_switchbox()
        + theme(
            figure_size=(10.5, max(2.6, 1.0 + 1.2 * n_scenarios)),
            axis_text=element_blank(),
            axis_ticks=element_blank(),
            axis_line=element_blank(),
            panel_grid=element_blank(),
            plot_title=(
                element_blank()
                if title_parts
                else element_text(size=11, ha="left", margin={"b": -8, "units": "pt"})
            ),
        )
    )
    for a in extra_annotations:
        p = p + a

    if title_parts:
        fig = p.draw()
        renderer = fig.canvas.get_renderer()
        x_pos = fig.axes[0].get_position().x0
        for text_str, color in title_parts:
            t = fig.text(
                x_pos, 0.92, text_str,
                color=color, fontsize=12, fontweight="bold",
                fontfamily="GT Planar", va="bottom",
            )
            bb = t.get_window_extent(renderer=renderer)
            x_pos = bb.transformed(fig.transFigure.inverted()).x1
        return fig

    return p

DELTA_META_COLS = [
    BLDG_ID, "weight", "postprocess_group.has_hp",
    HEATING_TYPE_COL, "heats_with_natgas", "heats_with_oil", "heats_with_propane",
]


def _lmi_delta_state(
    before_lf: pl.LazyFrame,
    after_lf: pl.LazyFrame,
    before_bill_cols: list[str],
    after_bill_cols: list[str],
) -> pl.DataFrame:
    """Compute statewide bill deltas using specified bill columns."""
    current = (
        before_lf
        .filter(pl.col("month") == ANNUAL_MONTH)
        .select(DELTA_META_COLS + before_bill_cols)
        .collect()
        .with_columns(
            pl.sum_horizontal(before_bill_cols).alias("bill_before"),
        )
    )
    after = (
        after_lf
        .filter(pl.col("month") == ANNUAL_MONTH)
        .select([BLDG_ID] + after_bill_cols)
        .collect()
        .with_columns(
            pl.sum_horizontal(after_bill_cols).alias("bill_after"),
        )
        .select(BLDG_ID, "bill_after")
    )
    return current.join(after, on=BLDG_ID, how="inner").with_columns(
        (pl.col("bill_after") - pl.col("bill_before")).alias("delta")
    )

Bill Alignment Test (BAT) results

The master BAT table contains per-building annual bill alignment values decomposed into delivery, supply, and total components. A positive BAT_percustomer value means the customer’s bill exceeds their marginal cost of service — they cross-subsidize other customers. A negative value means they pay less than their cost — others cross-subsidize them.

We look at all customers under current rates (upgrade 0 / run 1+2) to quantify how current rate structures misalign bills with cost of service. BAT_percustomer_total is the sum of delivery and supply components. The unified histograms show the full population; the faceted versions break the distribution down by heating type (fossil fuel, electric resistance, heat pump).

def plot_bat_histogram(
    df: pl.DataFrame,
    col: str = "BAT_percustomer_total",
    title: str = "BAT per-customer (total): HP households under current rates",
) -> ggplot:
    """Weighted histogram of BAT per-customer values."""
    x_lo = math.floor(_weighted_percentile(df, col, "weight", 0.01) / 500) * 500
    x_hi = math.ceil(_weighted_percentile(df, col, "weight", 0.99) / 500) * 500

    binned = (
        df.with_columns(
            ((pl.col(col) / BAT_BIN_WIDTH).floor() * BAT_BIN_WIDTH + BAT_BIN_WIDTH / 2)
            .alias("bin_center"),
        )
        .with_columns(
            pl.when(pl.col("bin_center") < 0)
            .then(pl.lit("underpaying"))
            .otherwise(pl.lit("overpaying"))
            .alias("sign"),
        )
        .group_by("bin_center", "sign")
        .agg(pl.col("weight").sum().alias("weight_sum"))
    )

    sign_dtype = pl.Enum(["underpaying", "overpaying"])
    plot_df = binned.with_columns(pl.col("sign").cast(sign_dtype))

    total_weight = df["weight"].sum()
    pct_over = df.filter(pl.col(col) >= 0)["weight"].sum() / total_weight * 100
    pct_under = 100 - pct_over
    w_mean = _weighted_mean(df, col, "weight")

    max_weight = cast(float, plot_df["weight_sum"].max()) if not plot_df.is_empty() else 0.0
    y_annot = max_weight * 1.08

    p = (
        ggplot(plot_df, aes(x="bin_center", y="weight_sum", fill="sign"))
        + geom_col(width=BAT_BIN_WIDTH * 0.9)
        + geom_vline(xintercept=0, linetype="solid", color="gray", size=0.8)

            ha="left" if w_mean >= 0 else "right",

        )
        + scale_fill_manual(values=BAT_COLORS)

        + guides(fill=False)
        + theme_switchbox()
        + theme(figure_size=(10.5, 4.5))
    )

    if pct_under >= 1:
        p = p + annotate(
            "text", x=(x_lo + min(0, x_hi)) / 2, y=y_annot,
            label=f"Underpaying\n{pct_under:.0f}%",

Revenue requirement composition

Each utility’s total residential revenue requirement — the amount it must recover from residential ratepayers — breaks into three pieces:

  1. Delivery (Rate Case) — the base T&D revenue requirement set in the utility’s most recent rate case.
  2. Delivery Top-Ups — additional delivery-side charges for policy programs (SBC, CES delivery, EV make-ready, DLM), DER credit recovery (VDER), and sunk-cost recovery (COVID arrears, late-payment waivers).
  3. Supply — energy commodity, merchant function charge, and CES supply surcharge.

Data comes from the computed revenue requirement YAMLs in rate_design/hp_rates/ny/config/rev_requirement/.

Revenue requirement vs. marginal cost allocation

The BAT decomposes each customer’s cost of service into marginal cost (economic burden: Σ load × MC) and residual cost (revenue requirement minus total marginal costs, allocated per customer). Because we run paired CAIRO simulations — delivery-only (run 1) and delivery+supply (run 2) — we can decompose both marginal cost and residual cost into delivery and supply components. This lets us compare the revenue requirement breakdown against the BAT’s cost allocation at a finer grain.

#| label: burden-compute
def burden_shares(
    bills_lf: pl.LazyFrame,
    bldg_ids: list,
    income_df: pl.DataFrame,
    bill_cols: list[str] = LMI_BILL_COLS,
    burden_threshold: float = BURDEN_THRESHOLD,
) -> tuple[float, float]:
    """Return (weighted_pct_below, weighted_pct_above) the burden threshold.

    bill_cols controls which columns are summed to form the total energy bill.
    Default uses the undiscounted columns; pass LMI_40_BILL_COLS or
    LMI_100_BILL_COLS to use LMI-discounted bills.
    """
    annual = (
        bills_lf
        .filter(pl.col("month") == ANNUAL_MONTH)
        .select(BLDG_ID, "weight", *bill_cols)
        .collect()
        .filter(pl.col(BLDG_ID).is_in(bldg_ids))
        .with_columns(
            pl.sum_horizontal(bill_cols).alias("total_bill"),
        )
        .join(income_df.select(BLDG_ID, INCOME_COL), on=BLDG_ID, how="inner")
        .with_columns(
            (pl.col("total_bill") / pl.col(INCOME_COL)).alias("burden"),
        )
        .with_columns(
            (pl.col("burden") > burden_threshold).alias("is_burdened"),
        )
    )
    total_weight = annual["weight"].sum()
    burdened_weight = annual.filter(pl.col("is_burdened"))["weight"].sum()
    pct_above = burdened_weight / total_weight * 100
    pct_below = 100 - pct_above
    return (pct_below, pct_above)

def plot_burden_bar(
    below_pcts: list[float],
    above_pcts: list[float],
    scenario_labels: list[str] = BURDEN_SCENARIO_LABELS,
):
    """Diverging bar chart of burden shares centered on the 6% threshold.

    Orange bars (above 6%) extend upward from y=0; blue bars (below 6%)
    extend downward. The y=0 line represents the burden threshold.
    """
    rows = []
    for scenario, above, below in zip(scenario_labels, above_pcts, below_pcts):
        rows.append({"scenario": scenario, "segment": "Above 6%",
                     "value": above, "label": f"{round(above)}%",
                     "label_y": above / 2})
        rows.append({"scenario": scenario, "segment": "Below 6%",
                     "value": -below, "label": f"{round(below)}%",
                     "label_y": -below / 2})

    df = pl.DataFrame(rows).with_columns(
        pl.col("scenario").cast(pl.Enum(scenario_labels)),
        pl.col("segment").cast(pl.Enum(BURDEN_SEGMENT_ORDER)),
    )

    return (
        ggplot(df, aes(x="scenario", y="value", fill="segment"))
        + geom_col(width=0.5)
        + geom_text(
            aes(y="label_y", label="label"),
            color="white", fontweight="bold", size=14, fontstyle="italic",
        )
        + geom_hline(yintercept=0, size=1.5, color="#333333")
